# src/dazzle/back/runtime/auth/secret_rotation.py
"""Connection secret-rotation helpers (#1342 — grace window + audit).

Pure, store-free: the grace-duration parser used by the rotate CLI, the audit
event-name constants, and the ``connection_secret_events`` DDL mirrored by the
auth store's ``_init_db`` (alongside alembic 0012). Kept separate from the DSL
lexer's ``parse_duration`` so the auth store doesn't depend on the core parser.
"""

from __future__ import annotations

import re
from datetime import timedelta

# Audit event names (written to connection_secret_events.event).
SECRET_EVENT_ROTATED = "rotated"
SECRET_EVENT_REVOKED_PREVIOUS = "revoked_previous"
SECRET_EVENT_KEY_REWRAPPED = "encryption_key_rewrapped"

_GRACE_UNITS = {"m": "minutes", "h": "hours", "d": "days", "w": "weeks"}
_GRACE_RE = re.compile(r"^([1-9][0-9]*)([mhdw])$")


def parse_grace_duration(text: str) -> timedelta:
    """``'24h'`` / ``'7d'`` / ``'30m'`` / ``'2w'`` -> timedelta.

    A single positive integer + one unit suffix (m/h/d/w). Raises ``ValueError``
    on anything else (empty, zero/negative, no unit, compound, fractional, 'y').
    """
    match = _GRACE_RE.match(text.strip())
    if match is None:
        raise ValueError(
            f"invalid grace duration {text!r} — use <positive int><m|h|d|w>, e.g. 24h"
        )
    amount = int(match.group(1))
    return timedelta(**{_GRACE_UNITS[match.group(2)]: amount})


CONNECTION_SECRET_EVENTS_DDL = """
CREATE TABLE IF NOT EXISTS connection_secret_events (
    seq           BIGINT GENERATED BY DEFAULT AS IDENTITY,
    id            TEXT PRIMARY KEY,
    connection_id TEXT NOT NULL,
    tenant_id     TEXT NOT NULL,
    event         TEXT NOT NULL,
    actor         TEXT,
    detail        TEXT NOT NULL DEFAULT '{}',
    at            TEXT NOT NULL
)
"""

CONNECTION_SECRET_EVENTS_INDEXES: tuple[str, ...] = (
    "CREATE INDEX IF NOT EXISTS ix_connection_secret_events_conn "
    "ON connection_secret_events(connection_id, seq)",
)

        self, connection_id: str, new_secrets: dict[str, Any], *,
        grace: "timedelta | None" = None, actor: str | None = None,
        tenant_id: str | None = None,

        """Rotate a connection's secret, optionally keeping the OLD one valid for
        ``grace`` (SCIM-bearer overlap). One transaction: writes the new blob, sets-or-
        clears the previous_* grace columns, bumps ``updated_at``, and appends a
        ``rotated`` audit event (non-secret detail). Returns True if a row changed."""
        import json

        from dazzle.back.runtime.auth.connection_crypto import encrypt_secret
        from dazzle.back.runtime.auth.secret_rotation import SECRET_EVENT_ROTATED

        now_dt = datetime.now(UTC)
        now = now_dt.isoformat()
        encrypted_new = encrypt_secret(json.dumps(new_secrets)) if new_secrets else None
        with self._transaction() as cur:
            if tenant_id is not None:
                cur.execute(
                    "SELECT encrypted_secret, tenant_id, type FROM connections "
                    "WHERE id = %s AND tenant_id = %s",
                    (connection_id, tenant_id),
                )

                cur.execute(
                    "SELECT encrypted_secret, tenant_id, type FROM connections WHERE id = %s",

                return False

            if grace is not None:
                expires = (now_dt + grace).isoformat()
                cur.execute(
                    "UPDATE connections SET encrypted_secret = %s, "
                    "previous_encrypted_secret = %s, previous_secret_expires_at = %s, "
                    "updated_at = %s WHERE id = %s",
                    (encrypted_new, row["encrypted_secret"], expires, now, connection_id),
